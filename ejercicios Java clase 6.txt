import java.util.Scanner;
public class OtraClase {

	public static void main(String[] args) {
	Scanner sc = new Scanner(System.in);
	System.out.println("ingrese 2 numeros para mostrar sus correspondientes...");
	Double uno = sc.nextDouble();
	Double dos = sc.nextDouble();
	
	Double suma = uno+dos;
	Double resta = uno-dos;
	Double multiDouble = uno*dos;
	Double diviDouble = uno/dos;
	
	System.out.println("La suma es:" + suma);
	System.out.println("La resta es:" +resta);
	System.out.println("La multiplicacion es:" +multiDouble);
	System.out.println("La division es:" + diviDouble);
	sc.close();
	}}
